# 预研目标定义模板
pre_research_goal:
  # 业务问题定义
  business_problem:
    - 当前痛点: "系统并发能力仅支持1万QPS，业务需求10万QPS"
    - 期望目标: "提升10倍并发能力，同时降低50%云服务成本"
    - 成功标准: "压测达到10万QPS，P99延迟<100ms"
  
  # 技术需求定义
  technical_requirements:
    - 功能性: "支持分布式事务、数据一致性"
    - 非功能性: "可用性99.99%、可观测性完善"
    - 约束条件: "必须兼容现有MySQL生态、团队学习成本<2周"
  
  # 预研范围边界
  scope_boundary:
    - 包含: "核心架构验证、性能基准测试、集成可行性"
    - 不包含: "完整功能开发、生产级部署、安全合规认证"

  technical_risks:
    - 风险项: "技术成熟度不足"
      可能性: "中"
      影响程度: "高"
      缓解措施: "选择有商业支持的版本"
      责任人: "架构师"
    
    - 风险项: "性能不达标"
      可能性: "低"
      影响程度: "高"
      缓解措施: "预留优化时间窗口"
      责任人: "性能工程师"
  
  # 集成风险
  integration_risks:
    - 风险项: "与现有系统兼容性问题"
      可能性: "中"
      影响程度: "中"
      缓解措施: "设计适配层"
      责任人: "技术负责人"
  
  # 人员风险
  people_risks:
    - 风险项: "团队学习曲线陡峭"
      可能性: "高"
      影响程度: "中"
      缓解措施: "安排专项培训、外部专家支持"
      责任人: "技术经理"

# 决策标准
decision_criteria:
  go_conditions:
    - "POC测试通过率 >= 90%"
    - "性能指标达成率 >= 95%"
    - "高风险项均有缓解措施"
    - "ROI预估 >= 200%"
  
  no_go_conditions:
    - "核心功能验证失败"
    - "性能差距 > 30%"
    - "存在无法缓解的高风险"
    - "成本超出预算50%"
